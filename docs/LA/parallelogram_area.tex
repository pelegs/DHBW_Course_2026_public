\begin{enumerate}
    \item $\area{\hat{\mathbf{x}}}{\hat{\mathbf{y}}} = 1$,
    \item $\area{\alpha\vvec{a}}{\beta\vvec{b}} = \alpha\beta\area{\vvec{a}}{\vvec{b}}$,
    \item $\area{\vvec{a}}{\alpha\vvec{a}} = 0$,
    \item $\area{\vvec{a}}{\vvec{b}} = -\area{\vvec{b}}{\vvec{a}}$,
    \item 
        \begin{enumerate}
            \item $\area{\vvec{a}}{\vvec{b}+\vvec{c}} = \area{\vvec{a}}{\vvec{b}} + \area{\vvec{a}}{\vvec{c}}$,
            \item $\area{\vvec{a}+\vvec{b}}{\vvec{c}} = \area{\vvec{a}}{\vvec{c}} + \area{\vvec{b}}{\vvec{c}}$.
        \end{enumerate}
\end{enumerate}
